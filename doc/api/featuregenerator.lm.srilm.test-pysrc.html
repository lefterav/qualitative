 1  #!/usr/bin/env python2.4 
 2   
 3  # Use the srilm module 
 4  from srilm import * 
 5   
 6  # Initialize a trigram LM variable (1 = unigram, 2 = bigram and so on) 
 7  n = initLM(3) 
 8   
 9  # Read 'sample.lm' into the LM variable 
10  readLM(n, "sample.lm") 
11   
12  # How many n-grams of different order are there ? 
13  print "There are", howManyNgrams(n, 1) , "unigrams in this LM" 
14  print "There are" , howManyNgrams(n, 2) , "bigrams in this LM" 
15  print "There are" , howManyNgrams(n, 3) , "trigrams in this LM" 
16  print 
17   
18  # Query the LM for some n-gram log probabilities. 
19  # Note that a SRI language model uses backoff smoothing, so if an n-gram is 
20  # not present in the LM, it will compute it using a smoothed lower-order  
21  # n-gram distribution. 
22  p1 = getUnigramProb(n, 'good') 
23  print "p('good') =", p1 
24  p2 = getBigramProb(n, 'nitin madnani') 
25  print "p('nitin madnani') =", p2 
26  p3 = getTrigramProb(n, 'there are some'); 
27  print "p('there are some') =", p3 
28  print 
29   
30  # Query the LM to get the final log probability for an entire sentence. 
31  # Note that this is  different from a n-gram probability because 
32    # (1) For a sentence, SRILM appends <s> and </s> to its beginning  
33    #     and the end respectively 
34    # (2) The log prob of a probability is the sum of all individual  
35    #     n-gram log probabilities 
36  sprob = getSentenceProb(n,'there are some good',4) 
37  print "p ('there are some good') =" , sprob 
38  print 
39   
40  # Query the LM to get the total log probability for the file named 'corpus' 
41  corpus = 'test.txt' 
42  corpus_prob = getCorpusProb(n, corpus) 
43  print "Logprob for the file" , corpus , "=" , corpus_prob 
44   
45  # Query the LM to get the perplexity for the file named 'corpus' 
46  corpus_ppl = getCorpusPpl(n, corpus); 
47  print "Perplexity for the file" , corpus , "=" , corpus_ppl 
48   
49  # Free LM variable 
50  deleteLM(n);
